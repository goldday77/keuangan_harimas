// Handler untuk ambil data (PULL) return jsonResponse({ transactions: getAllTransactions(), budgets: getAllBudgets() }); } function doPost(e) { const payload = JSON.parse(e.postData.contents); const action = payload.action; if (action === 'saveAll') { if (payload.transactions) saveAllTransactions(payload.transactions); if (payload.budgets) saveAllBudgets(payload.budgets); return jsonResponse({status: 'ok'}); } if (action === 'addTransaction') { appendTransaction(payload.transaction); return jsonResponse({status: 'ok'}); } } function getOrCreateSheet(name, headers) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(name); if (!sheet) { sheet = ss.insertSheet(name); sheet.appendRow(headers); } return sheet; } function getAllTransactions() { const sheet = getOrCreateSheet(SHEET_NAME_TX, ['id','type','amount','category','note','date']); const values = sheet.getDataRange().getValues(); if (values.length <= 1) return []; return values.slice(1).map(r => ({ id: String(r[0]), type: r[1], amount: r[2], category: r[3], note: r[4], date: r[5] })); } function saveAllTransactions(txs) { const sheet = getOrCreateSheet(SHEET_NAME_TX, ['id','type','amount','category','note','date']); sheet.clearContents(); sheet.appendRow(['id','type','amount','category','note','date']); if (txs.length > 0) { const rows = txs.map(t => [t.id, t.type, t.amount, t.category, t.note, t.date]); sheet.getRange(2, 1, rows.length, 6).setValues(rows); } } function getAllBudgets() { const sheet = getOrCreateSheet(SHEET_NAME_BUDGET, ['category','amount']); const values = sheet.getDataRange().getValues(); const budgets = {}; values.slice(1).forEach(r => { budgets[r[0]] = r[1]; }); return budgets; } function saveAllBudgets(b) { const sheet = getOrCreateSheet(SHEET_NAME_BUDGET, ['category','amount']); sheet.clearContents(); sheet.appendRow(['category','amount']); const rows = Object.keys(b).map(k => [k, b[k]]); if (rows.length > 0) sheet.getRange(2, 1, rows.length, 2).setValues(rows); } function jsonResponse(data) { return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON);
